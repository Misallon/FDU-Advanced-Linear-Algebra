\newcounter{problem}[section]
\renewcommand{\theproblem}{\arabic{problem}}
\newlength{\problemsep}
\setlength{\problemsep}{3.0\baselineskip}
\newcommand{\problemdivider}{%
  \par
  \vspace{\dimexpr\problemsep/2\relax}
  \hbox to \linewidth{\hfill\raisebox{0pt}[0pt][0pt]{\rule{0.35\linewidth}{0.4pt}}\hfill}%
  \vspace{\dimexpr\problemsep/2\relax}
}

\newenvironment{problem}[1][]{%
  \ifnum\value{problem}>0
    \problemdivider
  \fi
  \refstepcounter{problem}%
  \noindent\textbf{\theproblem.}%
  \if\relax\detokenize{#1}\relax\else\ \textbf{（#1）}\fi\ \ignorespaces}{%
  \par
}

\newenvironment{solution}{%
  \par\medskip
  \begingroup
  \setlength{\parindent}{0pt}
  \setlength{\parskip}{0.55em}
  \noindent{\large\bfseries Solution：}\par\smallskip
  \ignorespaces
}{%
  \par
  \endgroup
  \medskip
}

\newenvironment{solnote}{%
  \par\smallskip
  \begin{quote}
  \small\itshape\textbf{注：}\ignorespaces
}{%
  \end{quote}
  \smallskip
}

    \itemindent=0pt
  }%

  \itshape
  \ignorespaces
}{%
  \endlist
}
